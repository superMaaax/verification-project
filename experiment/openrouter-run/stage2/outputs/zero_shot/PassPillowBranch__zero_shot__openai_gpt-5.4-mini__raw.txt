class PassPillowBranch {
    /*@
      @ requires n > 1;
      @ requires time >= 0;
      @ ensures (\result >= 1 && \result <= n);
      @ ensures (\result == (((time % ((n - 1) * 2)) < n)
      @                    ? (time % ((n - 1) * 2)) + 1
      @                    : n * 2 - (time % ((n - 1) * 2)) - 1));
      @*/
    public int passPillow(int n, int time) {
        time = time % (n - 1) * 2;
        if (time < n) {
            return time + 1;
        }
        return n * 2 - time - 1;
    }
}
