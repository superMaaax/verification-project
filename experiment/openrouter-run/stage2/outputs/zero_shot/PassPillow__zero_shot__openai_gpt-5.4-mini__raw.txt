class PassPillow {
    /*@
      @ requires n >= 2;
      @ requires time >= 0;
      @ ensures \result >= 1 && \result <= n;
      @ ensures (\let int t = time % ((n - 1) * 2);
      @              (\result == (t < n ? t + 1 : n * 2 - t - 1)));
      @*/
    public int passPillow(int n, int time) {
        time %= (n - 1) * 2;
        return time < n ? time + 1 : n * 2 - time - 1;
    }
}
