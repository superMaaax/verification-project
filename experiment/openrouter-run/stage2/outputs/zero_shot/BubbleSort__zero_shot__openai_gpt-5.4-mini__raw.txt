public class BubbleSort { 
    
    /*@ public normal_behavior
      @   requires array != null;
      @   requires x >= 0 && x < array.length;
      @   requires y >= 0 && y < array.length;
      @   assignable array[x], array[y];
      @   ensures array[x] == \old(array[y]) && array[y] == \old(array[x]);
      @   ensures (\forall int i; i != x && i != y; array[i] == \old(array[i]));
      @ also
      @ public exceptional_behavior
      @   requires array == null || x < 0 || x >= array.length || y < 0 || y >= array.length;
      @   signals (Exception e) true;
      @*/
    void swap(int x, int y,  int array[]) {
        int temp;     
        temp = array[x];
        array[x] = array[y];
        array[y] = temp;
    }

    /*@ public normal_behavior
      @   requires arr != null;
      @   assignable arr[*];
      @   ensures \result == arr;
      @   ensures (\forall int i; 0 <= i && i < arr.length - 1; arr[i] <= arr[i+1]);
      @   ensures (\forall int i; 0 <= i && i < arr.length; (\exists int j; 0 <= j && j < \old(arr.length); arr[i] == \old(arr[j])));
      @*/
    int[] bubbleSort(int arr[]) {
        int n = arr.length;

        /*@ loop_invariant 0 <= i && i <= n - 1;
          @ loop_invariant (\forall int k; 0 <= k && k < i; 
          @                  (\forall int m; k < m && m < n; arr[k] <= arr[m]));
          @ decreases n - 1 - i;
          @*/
        for (int i = 0; i < n-1; i++) {	

            /*@ loop_invariant 0 <= j && j <= n - i - 1;
              @ loop_invariant (\forall int k; 0 <= k && k < j; 
              @                  arr[k] <= arr[k+1]);
              @ decreases n - i - 1 - j;
              @*/
            for (int j = 0; j < n-i-1; j++) {
                if (arr[j+1] < arr[j]) {  
		     swap(j, j + 1, arr); 
                } 
	    }
	} 
	return arr;
    } 
}
