class PowerOfFourBranch {
    /*@ public normal_behavior
      @ requires true;
      @ ensures \result <==> (\exists int k; k >= 0 && n == (1 << (2 * k)));
      @ ensures !\result <==> !(\exists int k; k >= 0 && n == (1 << (2 * k)));
      @*/
    public boolean isPowerOfFour(int n) {
        if(n <= 0) {
            return false;
        }
        if((n & (n - 1)) != 0) {
            return false;
        }
        if((n & 0xaaaaaaaa) != 0) {
            return false;
        }
        return true;
    }
}
