public class IntSquare {

    //@ requires x != Integer.MIN_VALUE;
    //@ ensures \result == x * x || \result == (-x) * (-x);
    public static int squareOf(int x) {
        if(x < 0)
            x = -x;
        int res = 0;
        //@ maintaining 0 <= i && i <= x;
        //@ maintaining res == i * x;
        //@ decreases x - i;
        for(int i = 0; i < x; i++) {
            //@ maintaining 0 <= j && j <= x;
            //@ maintaining res == i * x + j;
            //@ decreases x - j;
            for(int j = 0; j < x; j++) {
                res = res + 1;
            }
        }
        return res;
    }

}
