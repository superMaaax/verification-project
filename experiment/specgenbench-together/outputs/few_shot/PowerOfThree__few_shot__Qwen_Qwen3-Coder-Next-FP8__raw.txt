class PowerOfThree {
    //@ requires true;
    //@ ensures \result <==> (\exists int k; k >= 0 && n > 0; n == (int)Math.pow(3, k));
    public boolean isPowerOfThree(int n) {
        int temp = n;
        //@ maintaining temp == n / (3^(\at(\count, loop) - 1)) && temp != 0 && temp % 3 == 0;
        //@ decreases temp;
        while (temp != 0 && temp % 3 == 0) {
            temp /= 3;
        }
        return temp == 1;
    }
}
