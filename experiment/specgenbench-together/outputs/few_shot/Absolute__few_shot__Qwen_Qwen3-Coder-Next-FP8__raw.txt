public class Absolute {
	
	//@ ensures \result == num || \result == -num;
	//@ ensures \result >= 0;
	//@ ensures (\result == num) <==> (0 <= num);
	public  int Absolute(int num) {
		if (0 <= num)
			return num;
		else
			return -num;
	}

	
	//@ ensures \result == num || \result == -num;
	//@ ensures \result >= 0;
	//@ ensures (\result == num) <==> (0 <= num);
	public  long Absolute(long num) {
		if (0 <= num)
			return num;
		else
			return -num;	
	}
}
