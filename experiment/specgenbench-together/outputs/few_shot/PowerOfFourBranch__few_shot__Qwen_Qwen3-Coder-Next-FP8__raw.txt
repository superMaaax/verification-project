class PowerOfFourBranch {
    //@ ensures \result <==> (n > 0 && (\exists int k; k >= 0 && n == (int)Math.pow(4, k)));
    public boolean isPowerOfFour(int n) {
        if(n <= 0) {
            return false;
        }
        if((n & (n - 1)) != 0) {
            return false;
        }
        if((n & 0xaaaaaaaa) != 0) {
            return false;
        }
        return true;
    }
}
