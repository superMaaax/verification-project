class FizzBuzzSeq {
    //@ ensures res == 0 <==> !(n % 3 == 0 || n % 5 == 0);
    //@ ensures res == 3 <==> n % 3 == 0 && n % 5 != 0;
    //@ ensures res == 5 <==> n % 3 != 0 && n % 5 == 0;
    //@ ensures res == 8 <==> n % 3 == 0 && n % 5 == 0;
    public int fizzBuzz(int n) {
        int res = 0;
        res += ((n % 3 == 0) ? 3 : 0);
        res += ((n % 5 == 0) ? 5 : 0);
        return res;
    }
}
