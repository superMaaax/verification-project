class MyPower {
    //@ requires n >= 0;
    //@ ensures \result == (n == 0 ? 1 : (x == 0 ? 0 : \pow(x, n)));
    public static int power(int x, int n) {
        int res = 1;
        //@ maintaining res == \pow(x, i);
        //@ maintaining 0 <= i && i <= n;
        //@ decreases n - i;
        for(int i = 0; i < n; i++)
            res = res * x;
        return res;
    }
}
